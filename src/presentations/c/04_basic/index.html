                            scanf(“%f%f%f”,&a,&b,&c);
                            printf(“\n Co-efficients are a=%f\tb=%f\t c=%f”,a,b,c);
                            disc=b*b-4*a*c; /* Calculating discriminant */
                            if(disc == 0)
                            {
                           
                            printf(“\nRoots are real and equal”);
                            root1 = - b / (2.0*a);
                            root2=root1;
                            printf(“\n Root1=%f\t Root2=%f”,root1,root2);
                            }
                            else
                            {
                            if(disc<0)
                            {
                            printf(“\nRoots are complex”);
                            part_r = - b / (2.0*a);
                            part_i = sqrt(-disc) / (2.0*a);
                            printf(“\nRoot1=%f + i %f\t Root2 = %f - i%f”,part_r,part_i,part_r,part_i);
                            }
                            else
                           
                            {
                            printf( “\nRoots are real and distinct”);
                            root1 = (-b+sqrt(disc)) / (2.0*a);
                            root2 = (-b-sqrt(disc)) / (2.0*a)
                            printf(“\nRoot1=%f\tRoot2=%f”,root1,root2);
                            }
                            }
                            }
